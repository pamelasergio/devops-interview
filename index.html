Assume that the script's name is display.sh. Run the script with the following arguments: 
./display.sh hey there goodluck

		#!/bin/bash
		# (1) I am on the 1st line of the script, what am I?

		# (2) What's my output?
		echo "(2) $#"

		# (3) What's my output?
		echo "(3) $@"

		# (4) What's my purpose?
		echo "(4) $?"

		# (5) What's my output?
		echo "(5)"
		for i in $*; do
    		echo "$i"
		done
							
./display.sh hey there goodluck

		(2) 3
		(3) hey there goodluck
		(4) 0
		(5)
		hey
		there
		goodluck
